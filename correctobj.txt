```csharp
using UnityEngine;
using UnityEngine.XR.Interaction.Toolkit;

public class SocketCheck : MonoBehaviour
{
    public GameObject correctObject;       // Object that triggers the "Correct!" message
    public GameObject correctMessageUI;    // UI Text or Canvas to show

    private XRSocketInteractor socket;

    void Start()
    {
        socket = GetComponent<XRSocketInteractor>();
    }

    private void OnSelectEntered(XRBaseInteractable interactable)
    {
        if (interactable.gameObject == correctObject)
        {
            if (correctMessageUI)
                correctMessageUI.SetActive(true);
        }
    }

    private void OnSelectExited(XRBaseInteractable interactable)
    {
        if (interactable.gameObject == correctObject)
        {
            if (correctMessageUI)
                correctMessageUI.SetActive(false);
        }
    }
}
```
